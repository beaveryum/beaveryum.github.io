# 금액 원화 만단위부터 경까지 표시하기


금액을 만단위로해서 표시하기

예시 ) 199만9,990원

```php
function price_format($price){
    $memey_arr = ["", "만", "억", "조", "경"];
    $str = "";
    for($i = count($memey_arr) - 1; $i >= 0; --$i){
        $unit = pow(10000, $i);
        $part = floor($price / $unit);
        if($part > 0){
            $str .= number_format($part) . $memey_arr[$i];
        }
        $price %= $unit;
    }
    return $str."원";
}
```
