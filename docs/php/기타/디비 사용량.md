# 디비 사용량 과 용량 구하기  


```php
// 그누보드 전체 디비용량 구하기
$result = sql_query("show table status from ".G5_MYSQL_DB." like 'g5%'");
$db_size = 0;
while($dbData=sql_fetch_array($result)){
$db_size += $dbData['Data_length']+$dbData['Index_length'];
}
printf("%0.2f MB",$db_size / (1024*1024));

```
